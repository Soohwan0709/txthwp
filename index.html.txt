      stlite.mount(
        {
          requirements: ["python-docx"], // 필요한 라이브러리 자동 설치
          files: {
            "main.py": `
import streamlit as st
from docx import Document
from docx.shared import Pt
import io
import os

st.set_page_config(page_title="TXT to HWP", page_icon="📄")

# 디자인 개선
st.markdown("""
    <style>
    .main { background-color: #f0f2f6; }
    .stDownloadButton>button {
        width: 100%;
        border-radius: 8px;
        background-color: #007bff;
        color: white;
        border: none;
        padding: 10px;
    }
    </style>
    """, unsafe_allow_html=True)

def convert_to_hwp(content):
    doc = Document()
    style = doc.styles['Normal']
    style.font.name = '맑은 고딕'
    style.font.size = Pt(11)
    
    # 텍스트 줄바꿈 처리
    lines = content.decode("utf-8").splitlines()
    for line in lines:
        doc.add_paragraph(line)
    
    bio = io.BytesIO()
    doc.save(bio)
    return bio.getvalue()

st.title("📄 TXT to HWP 변환기")
st.info("터미널 설치 없이 브라우저에서 바로 작동하는 버전입니다.")

files = st.file_uploader("변환할 TXT 파일을 올려주세요 (여러 개 가능)", type=['txt'], accept_multiple_files=True)

if files:
    st.write(f"### 총 {len(files)}개의 파일")
    for f in files:
        with st.spinner(f'{f.name} 변환 중...'):
            data = convert_to_hwp(f.read())
            name = os.path.splitext(f.name)[0] + ".hwp"
            st.download_button(f"📥 {name} 다운로드", data=data, file_name=name)
            `,
          },
        },
        document.getElementById("root")
      );
    </script>
